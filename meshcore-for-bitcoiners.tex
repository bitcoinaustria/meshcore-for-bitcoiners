\begin{frame}{The one-line version}
  \begin{itemize}
    \item What Bitcoin did for money, MeshCore does for messaging:
      \textbf{a radio mesh you can join without asking}
    \item Generate a \textbf{keypair} and you're on --- no carrier, no number,
      no SIM, nobody to ask for permission
    \item Cheap radios relay each other in a \textbf{mesh}; coverage grows because
      people join, not because a company builds towers
  \end{itemize}
  \vfill
  \begin{alertblock}{Why a Bitcoiner should care}
    Spectrum is licensed, telcos are KYC gatekeepers, and the network itself is
    the chokepoint. MeshCore routes around all three ---
    \textbf{censorship-resistant, self-sovereign comms that still work when the
    grid is down}.
  \end{alertblock}
\end{frame}

\begin{frame}{Messaging without the network}
  \begin{itemize}
    \item Off-grid text messaging over long-range \textbf{LoRa} radio ---
      no internet, no cell tower, no SIM
    \item Open-source firmware running on cheap LoRa boards
    \item Uses \textbf{licence-free} ISM radio bands --- worldwide, region by
      region (EU 868\,MHz; US/AU 915\,MHz; \dots)
  \end{itemize}
  \vfill
  \begin{alertblock}{The EU angle}
    On these bands \emph{anyone} may transmit --- no licence, no registration, no
    fee. You only respect the power and duty-cycle limits.
  \end{alertblock}
\end{frame}

% The Bitcoiner's reflex after seeing it work: "so what's the catch?" Name the
% range/bandwidth trade-off now; the throughput bridge later pays it off.
\begin{frame}{The catch --- kilometres of range, bytes of bandwidth}
  \begin{itemize}
    \item \textbf{Long range, tiny pipe:} one \emph{hop} reaches \emph{tens} of
      kilometres line-of-sight (the packet above crossed Austria by chaining
      many) --- but at roughly the speed of a 1980s modem. One LoRa frame is at
      most \textbf{255 bytes}
    \item \textbf{Store-and-forward, not a phone call:} messages hop repeater to
      repeater with seconds of latency; you text, you don't stream
    \item \textbf{Physics, not policy:} the range buys reach by spending
      bandwidth and time --- and it sips power, so a node runs for days off a
      small battery or a solar panel
  \end{itemize}
  \vfill
  \begin{alertblock}{The right mental model}
    Not WhatsApp-over-radio --- think \textbf{a permissionless telegraph}: short
    text, long reach, no carrier. The narrow pipe is the price of needing nothing
    but a cheap radio and line of sight.
  \end{alertblock}
\end{frame}

% --------------------------------------------------------------- region codes
% The region "transport code" is an HMAC of the region key over the packet
% payload, truncated to 2 bytes; repeaters recompute it per hop to decide
% whether to relay (geofencing the flood). VERIFIED against MeshCore source
% (meshcore-decoder src/crypto/region-transport.ts; DeepWiki 7.6 "Region
% Filtering and Transport Codes"): region key = SHA256(regionName)[:16];
% code = LE-uint16 of the first 2 bytes of HMAC-SHA256(key, payloadType ||
% payload), with 0/0xFFFF nudged to 1/0xFFFE. Computed over the ENCRYPTED
% payload, so repeaters filter by region without decrypting. Symmetric and only
% 2 bytes wide => a routing/airtime filter, not a signature.
\fillerslide[Borders agreed by name, enforced per hop]{The region transport code}

% Plain-language setup before the bridge: the audience doesn't know MeshCore, so
% explain what a region code is and why it exists before drawing the SIGHASH bridge.
\begin{frame}{What is a region code, and why?}
  \begin{columns}[T]
    \begin{column}{0.66\textwidth}
      \begin{itemize}
        \item MeshCore \textbf{floods} adverts and path discovery. Unchecked,
          traffic from far away wastes everyone's scarce airtime
        \item A \textbf{region} is just an agreed name --- e.g.\ \textcolor{baRed}{\texttt{at}}
          for all of Austria, or \textcolor{baRed}{\texttt{at-west}} / \textcolor{baRed}{\texttt{at-ost}}
          / \textcolor{baRed}{\texttt{at-sued}} for its three macro-regions (right)
        \item Every packet carries a 2-byte \textbf{transport code}, derived from
          a hash of the region name and the message body; a repeater relays a
          packet only if the code matches its own region
        \item The mesh is thus \textbf{geofenced by agreement} --- no registry,
          no coordinates, no permission: just a name the locals share
      \end{itemize}
    \end{column}
    \begin{column}{0.32\textwidth}
      \centering
      \includegraphics[width=\linewidth]{austria-regions}\\[0.3em]
      {\scriptsize Austria's three MeshCore macro-regions ---
       \texttt{at-west} (west) / \texttt{at-ost} (east) / \texttt{at-sued} (south);
       cities nest below, e.g.\ \texttt{at-w} = Vienna\\
       region scheme: \href{https://meshcore-austria.at}{meshcore-austria.at}}
    \end{column}
  \end{columns}
  \vfill
  \begin{alertblock}{The neat part}
    The code is computed over the \emph{encrypted} payload, so repeaters filter by
    region \textbf{without decrypting} --- and because it \textbf{commits to the
    packet's bytes}, you can't lift it onto a different packet (a Bitcoin parallel
    we'll sharpen later).
  \end{alertblock}
\end{frame}

\fillerslide[The strongest parallel]{Identity is a keypair}

\bridgeslide{Identity is a keypair}{%
  \begin{itemize}
    \item \textbf{Self-generated keypair}, no issuer
    \item Address = \textbf{hash of the public key}
    \item \textbf{``Not your keys, not your coins''}
    \item Verify the address \textbf{out-of-band}
  \end{itemize}}{%
  \begin{itemize}
    \item \textbf{Self-generated keypair}, on-device
    \item Routing target = \textbf{public-key prefix}
    \item \textbf{``Not your keys, not your identity''}
    \item Verify keys via \textbf{QR / word list} (TOFU\footnote{Trust On First Use --- check a key once out-of-band, then pin it.})
  \end{itemize}}{%
  Two different tricks for the short ID: Bitcoin \textbf{hashes} the pubkey (hidden
  until you spend); MeshCore just \textbf{truncates} it to a prefix --- a routing
  shortcut, so it can collide (1\,$\to$\,1--3\,B). And the identity is deliberately
  \emph{sticky} (no HD rotation, no seed phrase) --- so \textbf{back it up}: lose
  the key and you lose the \emph{handle}.}

% ---------------------------------------------------------------- signing
\fillerslide[The elegant twist]{The signing model}

\bridgeslide{Opposite jobs for the same keypair}{%
  \begin{itemize}
    \item Content is \textbf{public} (on-chain)
    \item The key protects \textbf{authorization}
    \item No valid signature $\to$ the transaction \textbf{does not exist}
  \end{itemize}}{%
  \begin{itemize}
    \item Identity is \textbf{public} (the advert --- a node's signed
      ``here is my key, here I am'')
    \item The key protects \textbf{content}
    \item Adverts are \textbf{signed} (Ed25519)
    \item DMs are \textbf{encrypted} (X25519)
  \end{itemize}}{%
  But channels/rooms share one PSK\footnote{Pre-Shared Key --- one secret everyone
  in the channel holds.}, so a group message proves only that \emph{someone} in the
  channel sent it --- \textbf{not who}. Anyone with the PSK can read \emph{and}
  forge the sender name. DMs (X25519) and the advert (Ed25519) authenticate;
  \textbf{channel posts don't}.}

% Region code <-> SIGHASH (moved here from the intro region explanation): the
% 2-byte transport code commits to the (encrypted) payload, like a signature
% commits to its own tx. Audience already met region codes in the intro.
\bridgeslide{The region rides inside the message}{%
  \begin{itemize}
    \item A signature \textbf{commits to the transaction} it signs --- SIGHASH
      flags even pick \emph{which} parts it covers
    \item Lift it onto a different tx and it is invalid --- authorization is
      \textbf{bound to the content}
  \end{itemize}}{%
  \begin{itemize}
    \item The 2-byte region code is an \textbf{HMAC\footnote{Hash-based Message
      Authentication Code --- a keyed hash of the message.} committing to the
      payload}; repeaters recompute it per hop to decide whether to relay
    \item The region key is just \texttt{SHA256(name)[:16]} --- agree on
      ``\texttt{at-w}'' (Vienna), no key exchange, no registry
  \end{itemize}}{%
  No flags, no signer, no authorship: symmetric and \textbf{forgeable} (anyone who
  knows the region name can recompute it) and only 2 bytes wide --- a flood-scoping
  filter that saves airtime. It binds \emph{scope}, not \emph{authorship}.}

\fillerslide[A chosen constraint, not a weakness]{Throughput}

% EU 868 duty cycle VERIFIED (CEPT/ERC 70-03 + ETSI EN 300 220-2; implemented in
% Austria via the national frequency plan). MeshCore's EU default 869.525 MHz
% falls in the 869.4--869.65 MHz sub-band ("P"/g3) => 10% duty cycle, up to
% 500 mW (+27 dBm) ERP. (The lower 868.0--868.6 MHz sub-band is only 1% at
% 25 mW --- MeshCore deliberately picks the generous one.) Sources: The Things
% Network "Regional limitations of RF use"; rf.guru "Meshtastic, MeshCore,
% 868 MHz and the Ham Radio Trap"; disk91.com 868 MHz regulation overview.
% NOTE: still sanity-check against the current AT frequency plan before stage.
% Frequency: 869.525 MHz is MeshCore's documented EU/UK "Narrow" preset (the
% historical 869.525 RX2 frequency) and matches the SF8/BW62.5/CR4/8 below; an
% earlier draft said 869.618 MHz, which maps to no standard preset -- don't revert.
% Throughput numbers (estimation, both sides):
%  Bitcoin: ~2,000-3,000 tx/block, 1 block/10 min => ~3-5 tx/s ~= 200-300 tx/min.
%  MeshCore (EU/UK Narrow preset 869.525 MHz, SF8, BW62.5 kHz, CR 4/8): LoRa time-on-air
%  (Semtech ToA: 8-sym preamble, explicit header, CRC, DE=0) for a 16 B short
%  text ~0.25 s, 64 B ~0.64 s, 184 B full packet ~1.62 s; lighter coding 4/5
%  trims ~1/3 (16 B ~0.19 s ... 184 B ~1.05 s). 10% duty cycle = 6 s airtime/min/node
%  => ~24/min (short) down to ~4/min (full packet); ~30/min for tiny pings @4/5.
% SPEAKER: stage the airtime figure as a reveal --- ask "how much airtime per
% minute is a node legally allowed? ... six seconds." The smallness is the hook.
\bridgeslide{The throttle is imposed, not a failure}{%
  \begin{itemize}
    \item Block size is a \textbf{consensus} cap: one block every $\sim$10\,min
    \item $\sim$2{,}000--3{,}000 tx/block $\Rightarrow$ $\sim$3--5\,tx/s
      $\approx$ \textbf{200--300 tx/min}, network-wide
  \end{itemize}}{%
  \begin{itemize}
    \item Duty cycle is a \textbf{regulatory} cap: 10\% airtime on the
      869.4--869.65\,MHz sub-band (EU default: 869.525\,MHz, SF\,8, BW\,62.5\,kHz, CR\,4/8)
    \item One msg $\approx$ 0.25--1.6\,s on air (short text $\to$ full 184\,B);
      10\% $=$ 6\,s airtime/min $\Rightarrow$ \textbf{$\sim$4--24 msgs/min per node}
  \end{itemize}}{%
  Bitcoin's rate is one \emph{global} ledger; MeshCore's is \emph{per node} on a
  \textbf{shared} channel --- many nodes split the same airtime, so the real
  ceiling is lower. Either way the throttle is imposed, not ``they couldn't do
  more'' --- and a deliberate cap is exactly what forces a second layer to exist.}

% ---------------------------------------------------------------- lightning
\fillerslide[Two distinct analogies --- keep them separate]{Lightning Network}

\bridgeslide{Angle A --- route along a discovered path}{%
  \begin{itemize}
    \item \textbf{Route through specific channels} --- you pick the path
    \item Don't broadcast the payment \textbf{to the whole network}
  \end{itemize}}{%
  \begin{itemize}
    \item An advert floods once; \textbf{path info accumulates}
    \item Later DMs are \textbf{source-routed along that path}
  \end{itemize}}{%
  \begin{itemize}
    \item \textbf{Discovery differs:} Lightning gossips the \emph{whole} graph
      (proactive); MeshCore floods one advert to learn \emph{one} path (reactive)
    \item Lightning \textbf{onion-routes} (hop privacy); MeshCore path bytes ride
      in the clear
    \item Lightning carries \textbf{atomic value} (HTLCs); MeshCore carries data,
      no settlement
  \end{itemize}}

\begin{frame}{Angle B --- bootstrap, then jump layers}
  \begin{itemize}
    \item Move the bulk transfer to a \textbf{fat pipe} (internet, CB radio) --- off-mesh
    \item The mesh carries only the rare, \textbf{expensive anchoring event}, never the payload
    \item Key and rendezvous get negotiated over \textbf{encrypted DMs} --- on the mesh itself
  \end{itemize}
  \vfill
  \begin{alertblock}{Where it breaks}
    Lightning stays cryptographically chained to L1 (you can unilaterally
    close). The mesh\,→\,fat-pipe handoff is a \textbf{one-time bootstrap}, not
    a trust-anchored channel --- closer to WebRTC signaling or a BitTorrent
    tracker.
  \end{alertblock}
\end{frame}

% ---------------------------------------------------------------- channel hash
% A Bitcoiner spots SHA256-twice in the channel hash and reaches for "mining".
% Same primitive, different game --- make the distinction, then name the parallel
% that *does* hold (commitment + truncation trade-off).
% Construction VERIFIED against MeshCore source (firmware + michaelhart/
% meshcore-decoder, src/crypto/channel-crypto.ts; "Hitchhiker's Guide to
% MeshCore Cryptography"): for a "#room", key = SHA256("#room")[:16] (the
% AES-128 key), channel id = SHA256(key)[0] (first byte). Cipher is AES-128-ECB
% (+ HMAC-SHA256 MAC) --- NOT AES-256/CTR.
\fillerslide[Same primitive, a different game]{Double SHA256 --- but not mining}

\begin{frame}{Joining \texttt{\#bitcoin-at}: SHA256, twice, cropped}
  \texttt{SHA256(SHA256(\#name)[:16])} --- \textbf{hash} the channel
  \texttt{\#name}, \textbf{crop} to 16 bytes (the AES-128 key), \textbf{hash
  again}, take the first byte as a tiny public channel id. First pass: a
  \textbf{KDF} (key-derivation function, name\,$\to$\,AES-128 key); second: a
  commitment (a short id for that key). Two SHA256 rounds \emph{look} like
  Bitcoin --- but this is not mining.
  \vfill
  \centering\small
  \setlength{\tabcolsep}{8pt}\renewcommand{\arraystretch}{1.3}%
  \begin{tabular}{@{}lccc@{}}
    \toprule
      & \textcolor{baRed}{\textbf{MeshCore}} & \multicolumn{2}{c}{\textbf{Bitcoin}} \\
    \cmidrule(lr){3-4}
      & channel hash & txid / block id & \textbf{mining} \\
    \midrule
      Construction      & SHA256 + crop + SHA256 & SHA256d (full) & SHA256d (full) \\
      Computed          & once, fixed            & once, fixed    & \textbf{nonce search} \\
      Difficulty target & no                     & no             & \textbf{yes} \\
      Purpose           & id / commitment        & id / commitment & proof-of-work \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{alertblock}{The parallel that holds}
    No nonce search, no target, no contest --- so not mining. The real Bitcoin
    echo is \textbf{a short one-way commitment plus truncation as a namespace
    trade-off} --- like HASH160 addresses or txid prefixes, shorter is cheaper
    but collides sooner, the same tension as the 1\,B\,$\to$\,1--3\,B prefix.
  \end{alertblock}
\end{frame}

% Follows the "not mining" slide: a Bitcoiner asks whether LoRa has ASICs too.
% It does -- the LoRa PHY (chirp spread spectrum) is implemented in a dedicated
% radio IC, the Semtech SX1262 (the modem runs in hardware; the MCU just hands it
% bytes over SPI; ~uA in sleep). VERIFIED: Semtech SX1261/SX1262 datasheet/product
% pages; the Heltec LoRa 32 V3 in this deck uses an SX1262; ST's STM32WL packs the
% same SX126x radio IP onto an MCU die. The analogy BREAKS on the reason: a mining
% ASIC wins a competitive PoW race (more efficiency -> more reward, difficulty
% follows); the SX1262 is fixed-function and just saves power -- no contest.
\fillerslide[Same silicon idea, opposite reason]{ASICs --- and the LoRa radio chip}

% Float a small photo of the SX1262 high in the top-right corner of the
% bridgeslide --- small enough to sit up by the title/page number, not over the
% MeshCore column. (Bare name; resolves via the deck's pix/ graphics path.)
\cornerimage[width=0.10\paperwidth,topclear=20pt]{sx1262}
\bridgeslide{Purpose-built silicon, for opposite reasons}{%
  \begin{itemize}
    \item Mining \textbf{ASICs} race to compute SHA256 hashes
    \item Efficiency wins \textbf{reward} --- an arms race; more hashpower is
      always better, and difficulty just pulls up to match
  \end{itemize}}{%
  \begin{itemize}
    \item The \textbf{SX1262} radio chip runs the whole LoRa modem in hardware,
      not in software
    \item Dedicated silicon for \textbf{power}: microamps asleep, so a small
      coin-cell battery lasts for ages --- a general CPU couldn't really run the
      chirp-spread-spectrum modem at all, let alone that efficiently
  \end{itemize}}{%
  Both are purpose-built chips --- but Bitcoin's ASIC wins a \emph{contest} (more
  efficiency $\to$ more reward), while the SX1262 just \emph{sips power} at a
  fixed job. No race to win: the ceiling is physics and the duty cycle, not rival
  hashpower.}

% ---------------------------------------------------------- brainwallet rooms
% A #hashtag room derives its AES-128 key straight from the room name:
% key = SHA256("#" + name)[:16] (consistent with the channel-hash slide above).
% No salt, no stretching -> a brainwallet. VERIFIED against the MeshCore source
% and jacksbrain.com "Hitchhiker's Guide to MeshCore Cryptography": room names
% are lowercase alphanumerics + hyphens, <=30 chars; every name under 7 chars is
% exhaustively brute-forced in ~90 s on a 2023 laptop GPU (100M+ keys/s). A
% public WebGPU brute-forcer exists: github.com/jkingsman/meshcore-hashtag-cracker
% (added to the references slide). NOT an AES break -- AES-128 is untouched; the
% effective security is the entropy of the *name*. Hashtag rooms are world-
% readable *by design* (the honest divergence from the Bitcoin footgun).
\fillerslide[The name is the key --- a familiar footgun]{Hashtag rooms are brainwallets}

\begin{frame}{A hashtag room is a brainwallet}
  We just saw the room key \emph{is} \texttt{SHA256} of the name --- a Bitcoiner
  has seen exactly this go wrong.
  \vskip0.3em
  \centering
  \comparisontable{Bitcoin brainwallet}{MeshCore \#room}{%
    Key from           & \texttt{SHA256(passphrase)} & \texttt{SHA256(\#name)[:16]} \\
    Input              & low-entropy phrase          & short word (\texttt{\#vienna}) \\
    Salt / stretching  & none                        & none \\
    Public oracle      & the blockchain              & the broadcast packet \\
    Failure            & swept in seconds            & cracked in $\sim$90\,s ($<$7 chars) \\
  }
  \vfill
  \begin{alertblock}{The deep point --- a free verification oracle}
    Both hand the attacker a public way to \emph{check} an offline guess: the chain
    says ``this address holds coins''; the broadcast says ``this key decodes a
    message.'' Offline guessing \emph{plus} public verification is what makes both
    catastrophic --- \textbf{not} an AES break: the entropy is the \emph{name}.
  \end{alertblock}
\end{frame}

\begin{frame}{\dots\ but public \emph{by design}}
  \begin{itemize}
    \item \textbf{Lower stakes:} a brainwallet loses \emph{value}, irreversibly;
      a hashtag room loses \emph{confidentiality} (eavesdropping), not coins.
    \item \textbf{And it flatters MeshCore:} hashtag rooms are labelled
      \textbf{world-readable by design} --- ``no key to remember'' is the feature,
      not a hidden footgun. MeshCore \emph{tells} you the room is public.
    \item \textbf{Know which layer is public:} private traffic gets real crypto
      --- X25519\,/\,Ed25519 DMs, or private channels with long random hex keys.
  \end{itemize}
  \vfill
  \begin{alertblock}{The line}
    You wouldn't put your coins in \texttt{SHA256(``correct horse'')}. Don't put
    your secrets in \texttt{\#vienna}.
  \end{alertblock}
\end{frame}

% ============================ what Bitcoin's consensus buys, and MeshCore skips
% A four-slide section flowing from one insight: Bitcoin defends a scarce asset
% (coins), so it pays for heavy consensus machinery (PoW/Sybil resistance, one
% global order, permanent state); MeshCore defends nothing scarce -- it relays
% fleeting messages -- so it needs none of it. Each slide is one face of that.
\fillerslide[What is all that consensus machinery actually for?]{What Bitcoin's consensus buys --- and MeshCore skips}

% #1 Sybil resistance. (Per-topic divider dropped --- topic folded into the
% bridge title so this section isn't a stack of title cards.)
\bridgeslide{Sybil resistance --- what proof-of-work buys}{%
  \begin{itemize}
    \item PoW makes identity / participation deliberately \textbf{costly}
    \item That cost \textbf{is the security}: Sybil resistance, global ordering,
      permanent state
    \item You spend energy to earn a say
  \end{itemize}}{%
  \begin{itemize}
    \item Identity is \textbf{free} --- an Ed25519 keypair, generated offline, in
      unlimited number
    \item \textbf{No cost to participate} --- no proof-of-work
    \item Anyone can flood adverts
  \end{itemize}}{%
  \textbf{No Sybil resistance --- and none needed:} nothing to vote on, no scarce
  asset to seize. Abuse is only \emph{throttled}, never stopped --- \textbf{physical}
  limits (airtime / duty cycle, RF range) and \textbf{social} ones (repeater ACLs,
  admin passwords). The worst case is wasted airtime, not theft.}

% #2 Incentives. (Per-topic divider folded into the bridge title.)
\bridgeslide{Incentives --- paid security vs.\ a volunteer commons}{%
  \begin{itemize}
    \item Miners are \textbf{paid} --- block subsidy + fees
    \item Scarce block space $\to$ a \textbf{fee market} (a price)
    \item Incentives solve the free-rider problem
  \end{itemize}}{%
  \begin{itemize}
    \item Repeater operators are \textbf{unpaid} --- pure goodwill
    \item Scarce airtime $\to$ \textbf{social etiquette}: don't spam Public,
      use \#test, keep advert intervals conservative
    \item No price, no reward
  \end{itemize}}{%
  Bitcoin engineers incentives so the network sustains itself without trust;
  MeshCore \textbf{leans on volunteers and norms --- a commons with no fee
  market}. It works at today's scale, but nothing \emph{economic} defends it
  against the tragedy of the commons.}

% #3 State. (Per-topic divider folded into the bridge title.)
\bridgeslide{State --- consensus state vs.\ the moment}{%
  \begin{itemize}
    \item A \textbf{permanent, globally agreed ledger}
    \item Confirmed once $\to$ \textbf{there forever}, verifiable by anyone, later
  \end{itemize}}{%
  \begin{itemize}
    \item Channel messages exist \textbf{only as they are transmitted}
    \item Offline when it's sent $\to$ you miss it, silently, with no indication
    \item Room servers add \emph{optional} short-term history
  \end{itemize}}{%
  Bitcoin's whole purpose is durable shared state; MeshCore channels deliberately
  keep none. Different goals --- you can't ``re-sync'' a missed channel message
  the way a node replays the chain from genesis. (DMs can ACK, and
  identities/contacts do persist --- the contrast is \emph{channels}, not all of
  MeshCore.)}

% #4 Time -- ordering without a global clock. VERIFIED (see note below); the
% rule is scoped to ADVERTS, not arbitrary packets.
% VERIFIED against the MeshCore firmware behaviour via two sources independent of
% the "Hitchhiker's Guide" blog: DeepWiki's firmware annotation (ripplebiz/
% MeshCore 10.3 Troubleshooting) -- "Timestamp must be greater than
% last_advert_timestamp. If not, advert is rejected as possible replay attack"
% (src/helpers/BaseChatMesh.cpp) -- and the official FAQ (docs/faq.md), which
% documents the clock-not-set failure mode (a node with the wrong time stops
% appearing in discovery). IMPORTANT SCOPE FIX vs the brief: the forward-in-time
% rule is enforced on a node's *adverts* (self-announcements), keyed per pubkey
% by last_advert_timestamp -- NOT on every data packet. Slide reworded to say
% "adverts" so the claim stays watertight.
% #4 Time. (Per-topic divider folded into the bridge title.)
\bridgeslide{Time --- two ways to agree on ``after''}{%
  \begin{itemize}
    \item One chain orders \textbf{everything}, globally
    \item That total order is exactly what makes \textbf{double-spend impossible}
  \end{itemize}}{%
  \begin{itemize}
    \item A node's \textbf{adverts} must move \textbf{forward in time}: a repeater
      rejects one whose timestamp is $\le$ the last it saw for that key
      (\texttt{last\_advert\_timestamp}) --- a per-sender anti-replay rule
    \item Reset your radio with the wrong clock $\to$ your adverts are silently
      dropped and you stop appearing on the mesh
  \end{itemize}}{%
  Bitcoin builds one \emph{global} order (essential against double-spend);
  MeshCore enforces only \emph{per-sender} local order on adverts. There's no
  global ``which came first'' across senders --- with no asset to double-spend,
  it never needs the \emph{global} one.}

% ------------------------------------------------------------------- hard fork
% The firmware 1.14+ multi-byte routing prefix is backward-incompatible: a
% genuine "hard fork" framing --- with the honest caveat that there is no chain
% to actually split.
\fillerslide[Break the wire format, leave the laggards behind]{The multi-byte routing hard fork}

\bridgeslide{A backward-incompatible rule change}{%
  \begin{itemize}
    \item A \textbf{hard fork} changes the rules so old software rejects the new
      format
    \item Both sides can stay alive --- \textbf{two live chains}, each kept up by
      its own economic backers
  \end{itemize}}{%
  \begin{itemize}
    \item Firmware \textbf{1.14+} widens the routing prefix from 1 byte to
      1--3 bytes
    \item Old firmware cannot parse the new adverts / path hashes --- you must
      upgrade to stay on the mesh
  \end{itemize}}{%
  No consensus, no ledger, no coins --- nothing keeps a second version alive, so
  old firmware just \textbf{falls off the mesh} until it upgrades. Closer to a
  \textbf{TLS 1.2\,$\to$\,1.3} bump than a chain split: ``hard fork'' names the
  backward incompatibility, not a persistent divergence you could mine on.}

% ===================================================== shortcomings (Part 4) DRAFT
% DRAFT section --- "shortcomings from a Bitcoiner's perspective". Pulls the
% sharpest breaks out of the per-parallel callouts into one deliberate beat, and
% adds the items the parallels never reached (metadata leakage, no forward secrecy,
% default-channel weak key). Split deliberately into "deliberate by design" vs
% "real gaps" so the honest contract lands without sounding like a teardown.
% Sources for the added facts: jacksbrain.com "Hitchhiker's Guide to MeshCore
% Cryptography" (static-ECDH DMs -> no forward secrecy; channel/#room key model),
% localmesh.nl "MeshCore encryption details" (AES-128, default public key), and
% docs.meshcore.io. NOTE: keep an eye on 4:3 fit --- if either consolidation slide
% overflows, reach for \narrowonly{\small} (theme escape hatch).
\fillerslide[The honest reckoning --- what to still distrust]{Shortcomings, as a Bitcoiner sees them}

% Slide 1: the genuine gaps a privacy-minded Bitcoiner cares about most.
\begin{frame}{What still leaks --- even with encryption}
  \begin{itemize}
    \item \textbf{Metadata is in the clear:} adverts broadcast your pubkey and
      rough position, and source-routed paths expose \emph{who relays to whom} ---
      the social graph and location leak even when DM \emph{content} is encrypted
    \item \textbf{No onion routing:} repeaters must read the path prefixes to
      forward, so there is no hop privacy --- unlike Lightning's onion
    \item \textbf{No forward secrecy:} DMs use \emph{static} X25519 to your
      long-term key --- capture that key once and every \emph{recorded past} DM
      decrypts (a Signal-style ratchet is what's missing)
    \item \textbf{Sticky identity widens the blast radius:} the deliberately
      non-rotating key means one compromise exposes \emph{all} of your history
  \end{itemize}
  \vfill
  \begin{alertblock}{The honest line}
    Encryption protects \textbf{what you said}, not \textbf{that you said it, to
    whom, from where} --- and not retroactively if your key ever leaks.
  \end{alertblock}
\end{frame}

% Slide 2: the structural omissions --- mostly deliberate (callbacks), one real.
\begin{frame}{What nothing defends --- and why that's (mostly) on purpose}
  \begin{itemize}
    \item \textbf{No Sybil resistance:} identities are free, so spam is only
      \emph{throttled} by airtime and ACLs --- fine, because there's no scarce
      asset to seize
    \item \textbf{A volunteer commons:} repeaters are unpaid goodwill --- no fee
      market, so nothing \emph{economic} defends the network at scale
    \item \textbf{The default \#public channel ships a well-known key:} everyone
      has it --- convenience, \emph{not} privacy
    \item \textbf{Prefix squatting is cheap:} short routing prefixes are grindable
      --- a nuisance to \emph{discovery}, \emph{not} impersonation (full key
      $+$ TOFU is the backstop)
  \end{itemize}
  \vfill
  \begin{alertblock}{Deliberate vs.\ a real gap}
    Most of these are \textbf{features of defending nothing scarce} --- the worst
    case is wasted airtime, not theft. The two worth fixing before you trust it
    with anything sensitive: \textbf{metadata privacy} and \textbf{forward secrecy}.
  \end{alertblock}
\end{frame}

% Optional deep-dive on the on-wire packet format. Kept AFTER the closing as a
% backup slide so it can be skipped for a non-technical audience and shown only
% on demand --- it's self-contained, so skipping it breaks nothing in the main
% talk. It also ties several earlier slides together (region code, the 1--3 byte
% routing prefix, AES-128 + 2-byte MAC). [noframenumbering] keeps it out of the
% main slide count. VERIFIED against docs.meshcore.io/packet_format, DeepWiki 7.1
% "Packet Structure and Types", and michaelhart/meshcore-decoder.
\begin{frame}[noframenumbering]{Appendix 1: Anatomy of a MeshCore packet}
  \small
  \texttt{[\,header\,][\,transport codes?\,][\,path\_len\,][\,path\,][\,payload\,]}
  \hfill at most \textbf{255\,bytes} --- one LoRa frame
  \medskip
  \begin{itemize}
    \item \textbf{header} (1\,B) --- 2 bits route type, 4 bits payload type
      (advert, txt, group-text, ack, \dots), 2 bits version
    \item \textbf{transport codes} (4\,B, optional) --- only on \emph{transport}
      routes; where the 2-byte \textbf{region code} rides
    \item \textbf{path} (length byte $+$ up to 64\,B) --- \textbf{the core idea:}
      each relay \textbf{appends its own short ID} (1--3\,bytes) as it forwards
    \item \textbf{payload} (up to 184\,B) --- type-specific: an \emph{advert} is
      Ed25519-signed; a \emph{group / text} message is channel-hash $+$ 2-byte
      MAC $+$ AES-128 ciphertext
  \end{itemize}
  \vfill
  \begin{alertblock}{This is what makes it routing, not flooding}
    On a flood the path \emph{records} the relays a packet crossed; a direct
    message then \emph{carries that exact list} of router IDs, so only those
    repeaters forward it --- not every node, the way Meshtastic floods. The path
    \emph{is} the route.
  \end{alertblock}
  {\footnotesize Skip unless the room likes wire formats. Sources:
  \texttt{docs.meshcore.io/packet\_format}, DeepWiki 7.1, \texttt{meshcore-decoder}.}
\end{frame}
